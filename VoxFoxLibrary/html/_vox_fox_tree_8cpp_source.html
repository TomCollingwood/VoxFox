719 void VoxFoxTree::importObj(ngl::Obj * _mesh, ngl::Image * _texture,float _size, bool _normals, bool _interpnormals, bool _noCol,bool _shader)
720 {
721  std::vector<ngl::Vec3> verts = _mesh->getVertexList();
722  std::vector<ngl::Face> objFaceList = _mesh->getFaceList();
723  //std::vector<ngl::Vec3> normalList = _mesh->getNormalList();
724 
725  _mesh->calcDimensions();
726 
727  ngl::BBox boundingBox = _mesh->getBBox();
728  float meshheight = boundingBox.height();
729  float scale = _size/meshheight;
730  ngl::Vec3 centre = boundingBox.center();
731 
732 
733  std::vector<ngl::Vec3> vertNormals = std::vector<ngl::Vec3>(verts.size());
734  std::fill(vertNormals.begin(), vertNormals.end(), ngl::Vec3(0.0f,0.0f,0.0f));
735  std::vector<int> numberOfFacesPerVert = std::vector<int>(verts.size());
736  std::fill(numberOfFacesPerVert.begin(), numberOfFacesPerVert.end(), 0);
737 
738  std::vector<ngl::Vec3> vertTex = _mesh->getTextureCordList();
739 
740  if(_interpnormals && _normals)
741  {
742  for(int i = 0; i<objFaceList.size(); ++i)
743  {
744  ngl::Vec3 tmpNormal;
745 
746  tmpNormal = (verts[objFaceList[i].m_vert[1]]-verts[objFaceList[i].m_vert[0]]).cross(verts[objFaceList[i].m_vert[2]]-verts[objFaceList[i].m_vert[0]]);
747  tmpNormal.normalize();
748  for(int j =0; j<objFaceList[i].m_vert.size(); ++j)
749  {
750  vertNormals[objFaceList[i].m_vert[j]] += tmpNormal;
751  numberOfFacesPerVert[objFaceList[i].m_vert[j]]++;
752  }
753  }
754 
755  for(int i =0; i<(int)vertNormals.size(); ++i)
756  {
757  vertNormals[i]= vertNormals[i]/numberOfFacesPerVert[i] ;
758  vertNormals[i].normalize();
759  }
760  }
761 
762  for(std::vector<ngl::Face>::iterator itr=objFaceList.begin(); itr!=objFaceList.end(); ++itr)
763  {
764  if(itr->m_vert.size()==3)
765  {
766 
767  //-------------------------VERTEXES----------------------------
768  ngl::Vec3 a, b, c, e1;
769  a = verts[itr->m_vert[0]]*scale - centre;
770  b = verts[itr->m_vert[1]]*scale - centre;
771  c = verts[itr->m_vert[2]]*scale - centre;
772  e1 = b - a;
773 
774  ngl::Vec3 ta, tb, tc;
775  ta = vertTex[itr->m_tex[0]];
776  tb = vertTex[itr->m_tex[1]];
777  tc = vertTex[itr->m_tex[2]];
778 
779 
780  //--------------------------NORMALS-----------------------------
781  ngl::Vec3 na, nb, nc;
782  ngl::Vec3 tmpNormal;
783  if(_interpnormals && _normals)
784  {
785  na = vertNormals[itr->m_vert[0]];
786  nb = vertNormals[itr->m_vert[1]];
787  nc = vertNormals[itr->m_vert[2]];
788  }
789  else if(_normals)
790  {
791  tmpNormal = (verts[itr->m_vert[1]]-verts[itr->m_vert[0]]).cross(verts[itr->m_vert[2]]-verts[itr->m_vert[0]]);
792  tmpNormal.normalize();
793  }
794 
795  int steps = std::ceil(e1.length()/m_voxUnit);
796  ngl::Vec3 vecStep = e1/(2*steps); // scaled
797 
798  for(int i =0; i<(steps+1)*2; ++i) //scaled
799  {
800  ngl::Vec3 pos = a+(vecStep*i);
801  ngl::Vec3 posnormal = ngl::lerp(na,nb,((float)i)/((float)(((steps+1)*2)-1)));
802  ngl::Vec3 postexture = ngl::lerp(ta,tb,((float)i)/((float)(((steps+1)*2)-1)));
803  //ngl::Vec2 posUV = ;
804  ngl::Vec3 line = c-pos;
805  int jsteps = std::ceil(line.length()/m_voxUnit);
806 
807  if(i!=steps)
808  {
809  if(i%2==1) jsteps/=2;
810  else if(i%4==2) jsteps*=0.75;
811  else if(i%8==4) jsteps*=0.875;
812  }
813 
814  ngl::Vec3 mystep = line*(m_voxUnit / (3*line.length())); //scaled
815 
816  for(int j = 0; j<jsteps*3; ++j) //scaled
817  {
818  float inNX, inNY, inNZ;
819  float inU, inV;
820  if(_interpnormals && _normals)
821  {
822  ngl::Vec3 posnormal2 = lerp(posnormal,nc,((float)j)/((float)((jsteps*3) - 1)));
823  inNX = posnormal2.m_x;
824  inNY = posnormal2.m_y;
825  inNZ = posnormal2.m_z;
826  }
827  else if(_normals)
828  {
829  inNX = tmpNormal.m_x;
830  inNY = tmpNormal.m_y;
831  inNZ = tmpNormal.m_z;
832  }
833  Voxel toinsert = Voxel(inNX,inNY,inNZ,inU,inV);
834  if(!_noCol && _shader)
835  {
836  // Get RGB value from texture
837  // Put in shader as FragmentColor
838  ngl::Vec3 postexture2 = lerp(postexture,tc,((float)j)/((float)((jsteps*3) - 1)));
839  inU = postexture2.m_x;
840  inV = postexture2.m_y;
841  toinsert.r = (float)_texture->getColour(inU,inV).m_r/255.0f;
842  toinsert.g = (float)_texture->getColour(inU,inV).m_g/255.0f;
843  toinsert.b = (float)_texture->getColour(inU,inV).m_b/255.0f;
844  }
845  else if(_shader)
846  {
847  // Tells shader not to use FragmentColor
848  toinsert.r=2.0f;
849  }
850 
851  // We retain UVs if not using my specific shader
852  if(_shader)
853  {
854  toinsert.u=2.0f;
855  if(_normals) toinsert.v=2.0f;
856  }
857  else
858  {
859  toinsert.u=inU;
860  toinsert.v=inV;
861  }
862  addVoxel(glm::vec3(pos.m_x,pos.m_y,pos.m_z),toinsert);
863  pos = pos + mystep;
864  }
865  }
866  }
867  }
